## RECOMMENDED FIXES

### Script 1: Add Missing Tags

```python
#!/usr/bin/env python3
import re
from pathlib import Path

posts_dir = Path("/Users/yong/Desktop/personal/tech-blog/_posts")

for post_file in posts_dir.glob("*.md"):
    content = post_file.read_text()
    
    # Check if tags field exists
    if re.search(r'^tags:', content, re.MULTILINE):
        continue  # Already has tags
    
    # Parse front matter
    match = re.match(r'^---\n(.*?)\n---\n', content, re.DOTALL)
    if not match:
        continue
    
    fm = match.group(1)
    
    # Extract category
    category_match = re.search(r'^categories?:\s*\n?\s*-\s*(\w+)', fm, re.MULTILINE)
    default_category = category_match.group(1) if category_match else "general"
    
    # Generate default tags from filename
    filename = post_file.stem
    default_tags = [default_category]
    
    if "security" in filename.lower():
        default_tags.append("security")
    if "cloud" in filename.lower() or "aws" in filename.lower():
        default_tags.append("cloud")
    if "kubernetes" in filename.lower() or "docker" in filename.lower():
        default_tags.append("kubernetes")
    
    # Add tags field before image field
    new_fm = fm.replace(
        "image:",
        f"tags: {default_tags}\nimage:"
    )
    
    # Write back
    new_content = content.replace(f"---\n{fm}\n---", f"---\n{new_fm}\n---")
    post_file.write_text(new_content)
    print(f"Added tags to {post_file.name}")
```
